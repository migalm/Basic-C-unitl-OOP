int main(){
    /** MATRICES
            C0  C1  C2

    int A[][3]={{1,2,3},{4,5,6},{7,8,9}};
    int filas=3,columnas=3;

    for(int i=0;i<filas;i++){
        for(int j=0;j<columnas;j++){
